  8 /**
  9  * HasherEvent Object.
 10  * <p>According to the HTML5 spec `hashchange` event should have `oldURL` and `newURL` properties, since the only portion of the URL that changes is the hash I decided to use `oldHash` and `newHash` instead. (http://www.whatwg.org/specs/web-apps/current-work/multipage/history.html#event-hashchange)</p>
 11  * @param {String} eType	Hasher Event type.
 12  * @param {String} oldHash	Previous Hash.
 13  * @param {String} newHash	Current Hash.
 14  * @constructor
 15  */
 16 var HasherEvent = function(eType, oldHash, newHash){
 17 	/**
 18 	 * Event Type
 19 	 * @type String
 20 	 */
 21 	this.type = eType;
 22 	/**
 23 	 * Previous Hash value
 24 	 * @type String
 25 	 */
 26 	this.oldHash = oldHash;
 27 	/**
 28 	 * Current Hash value
 29 	 * @type String
 30 	 */
 31 	this.newHash = newHash;
 32 };
 33 
 34 /**
 35  * Returns string representation of the HasherEvent
 36  * @return {String} A string representation of the object.
 37  */
 38 HasherEvent.prototype.toString = function(){
 39 	return '[HasherEvent type="'+ this.type +'" oldHash="'+ this.oldHash +'" newHash="'+ this.newHash +'"]';
 40 };
