  9 /**
 10  * @namespace
 11  * @ignore
 12  */
 13 this.MM = this.MM || {};
 14 
 15 /**
 16  * EventDispatcher Object
 17  * @constructor
 18  */
 19 MM.EventDispatcher = function(){
 20 	this._handlers = {};
 21 };
 22 
 23 MM.EventDispatcher.prototype = {
 24 	
 25 	/**
 26 	 * Add Event Listener
 27 	 * @param {String} eType	Event Type.
 28 	 * @param {Function} fn	Event Handler.
 29 	 */
 30 	addEventListener : function(eType, fn){
 31 		if(typeof this._handlers[eType] == 'undefined'){
 32 			this._handlers[eType] = [];
 33 		}
 34 		this._handlers[eType].push(fn);
 35 	},
 36 	
 37 	/**
 38 	 * Remove Event Listener
 39 	 * @param {String} eType	Event Type.
 40 	 * @param {Function} fn	Event Handler.
 41 	 */
 42 	removeEventListener : function(eType, fn){
 43 		if(! this.hasEventListener(eType)){
 44 			return;
 45 		}
 46 		var	typeHandlers = this._handlers[eType], //stored for performance
 47 			n = typeHandlers.length;
 48 		if(n == 1){
 49 			this._handlers[eType] = null; //avoid loop if not necessary
 50 		}else{
 51 			while(n--){ //faster than for
 52 				if(typeHandlers[n] == fn){
 53 					typeHandlers.splice(n, 1);
 54 					break;
 55 				}
 56 			}
 57 		}
 58 	},
 59 	
 60 	/**
 61 	 * Removes all Listeners from the EventDispatcher object.
 62 	 */
 63 	removeAllEventListeners : function(){
 64 		this._handlers = {};
 65 	},
 66 	
 67 	/**
 68 	 * Checks if the EventDispatcher has any listeners registered for a specific type of event. 
 69 	 * @param {String} eType	Event Type.
 70 	 * @return {Boolean}
 71 	 */
 72 	hasEventListener : function(eType){
 73 		return (typeof this._handlers[eType] != 'undefined');
 74 	},
 75 
 76 	/**
 77 	 * Dispatch Event
 78 	 * - Call all Handlers Listening to the Event.
 79 	 * @param {Event|String} e	Custom Event Object (property `type` is required) or String with Event type.
 80 	 */
 81 	dispatchEvent : function(e){
 82 		e = (typeof e == 'string')? {type: e} : e; //create Object if not an Object to always call handlers with same type of argument.
 83 		if(this.hasEventListener(e.type)){
 84 			var typeHandlers = this._handlers[e.type], //stored for performance
 85 				curHandler;
 86 			for(var i=0, n = typeHandlers.length; i<n; i++){
 87 				curHandler = typeHandlers[i];
 88 				curHandler(e);
 89 			}	
 90 		}
 91 	}
 92 	
 93 };
