 23 /**
 24  * EventDispatcher Object
 25  * @constructor
 26  */
 27 MM.EventDispatcher = function(){
 28 	this._handlers = {};
 29 };
 30 
 31 MM.EventDispatcher.prototype = {
 32 	
 33 	/**
 34 	 * Add Event Listener
 35 	 * @param {String} eType	Event Type.
 36 	 * @param {Function} fn	Event Handler.
 37 	 */
 38 	addEventListener : function(eType, fn){
 39 		if(typeof this._handlers[eType] == 'undefined'){
 40 			this._handlers[eType] = [];
 41 		}
 42 		this._handlers[eType].push(fn);
 43 	},
 44 	
 45 	/**
 46 	 * Remove Event Listener
 47 	 * @param {String} eType	Event Type.
 48 	 * @param {Function} fn	Event Handler.
 49 	 */
 50 	removeEventListener : function(eType, fn){
 51 		if(! this.hasEventListener(eType)){
 52 			return;
 53 		}
 54 		var	typeHandlers = this._handlers[eType], //stored for performance
 55 			n = typeHandlers.length;
 56 		if(n == 1){
 57 			delete this._handlers[eType];
 58 		}else{
 59 			while(n--){ //faster than for
 60 				if(typeHandlers[n] == fn){
 61 					typeHandlers.splice(n, 1);
 62 					break;
 63 				}
 64 			}
 65 		}
 66 	},
 67 	
 68 	/**
 69 	 * Removes all Listeners from the EventDispatcher object.
 70 	 */
 71 	removeAllEventListeners : function(){
 72 		this._handlers = {};
 73 	},
 74 	
 75 	/**
 76 	 * Checks if the EventDispatcher has any listeners registered for a specific type of event. 
 77 	 * @param {String} eType	Event Type.
 78 	 * @return {Boolean}
 79 	 */
 80 	hasEventListener : function(eType){
 81 		return (typeof this._handlers[eType] != 'undefined');
 82 	},
 83 
 84 	/**
 85 	 * Dispatch Event
 86 	 * - Call all Handlers Listening to the Event.
 87 	 * @param {Event|String} e	Custom Event Object (property `type` is required) or String with Event type.
 88 	 */
 89 	dispatchEvent : function(e){
 90 		e = (typeof e == 'string')? {type: e} : e; //create Object if not an Object to always call handlers with same type of argument.
 91 		if(this.hasEventListener(e.type)){
 92 			var typeHandlers = this._handlers[e.type], //stored for performance
 93 				curHandler,
 94 				i,
 95 				n = typeHandlers.length;
 96 			e.target = e.target || this; //ensure Event.target exists (default to Object that extends EventDispatcher)
 97 			for(i=0; i<n; i++){
 98 				curHandler = typeHandlers[i];
 99 				curHandler(e);
100 			}	
101 		}
102 	}
103 	
104 };

119 /**
120  * Utilities for query string manipulation.
121  * @namespace
122  */
123 MM.queryUtils = {
124 	
125 	/**
126 	 * Gets full query as string with all special chars decoded.
127 	 * @param {String} [url]	 URL to be parsed, defaults to `location.href`
128 	 * @return {String}	Query string
129 	 */
130 	getQueryString : function(url){
131 		url = url || location.href; //used to avoid bug on IE6 and query string inside location.hash
132 		url = url.replace(/#.*/, ''); //removes hash (to avoid getting hash query)
133 		var queryString = /\?[a-zA-Z0-9\=\&\%\$\-\_\.\+\!\*\'\(\)\,]+/.exec(url); //valid chars according to: http://www.ietf.org/rfc/rfc1738.txt
134 		return (queryString)? decodeURIComponent(queryString[0]) : '';
135 	},
136 	
137 	/**
138 	 * Gets query as Object.
139 	 * - Alias for `MM.queryUtils.toQueryObject( MM.queryUtils.getQueryString() )`
140 	 * @return {Object}	Object with all the query "params => values" pairs.
141 	 */
142 	getQueryObject : function(){
143 		return this.toQueryObject(this.getQueryString());
144 	},
145 	
146 	/**
147 	 * Convert Query String into an Object
148 	 * @param {String} queryString	 Query String to be parsed
149 	 * @return {Object}	Object with all the query "params => values" pairs.
150 	 */
151 	toQueryObject : function(queryString){
152 		var queryArr = queryString.replace('?', '').split('&'), 
153 			n = queryArr.length,
154 			queryObj = {};
155 		while (n--) {
156 			queryArr[n] = queryArr[n].split('=');
157 			queryObj[queryArr[n][0]] = queryArr[n][1];
158 		}
159 		return queryObj;
160 	},
161 	
162 	/**
163 	 * Get query parameter value.
164 	 * @param {String} param	Parameter name.
165 	 * @param {String} [url]	URL to be parsed, default to location.href
166 	 * @return {String}	Parameter value.
167 	 */
168 	getParamValue : function(param, url){
169 		var queryObj = (url)? this.toQueryObject(url) : this.getQueryObject();
170 		return queryObj[param];
171 	},
172 	
173 	/**
174 	 * Checks if query contains parameter.
175 	 * @param {String} param	Parameter name.
176 	 * @param {String} [url]	URL to be parsed, default to location.href
177 	 * @return {Boolean} If parameter exist.
178 	 */
179 	hasParam : function(param, url){
180 		var regexp = new RegExp('(\?|\&)'+ param +'\=', 'g'); //matches `?param=` or `¶m=`
181 		return regexp.test(this.getQueryString(url));
182 	},
183 	
184 	/**
185 	 * Converts object into query string.
186 	 * @param {Object} obj	Object with "params => values" pairs.
187 	 * @return {String}	Formated query string starting with '?'.
188 	 */
189 	toQueryString : function(obj){
190 		var query = [],
191 			param;
192 		for(param in obj){
193 			query.push(param +'='+ obj[param]);
194 		}
195 		return (query.length)? '?'+ query.join('&') : '';
196 	}
197 	
198 };

212 		_checkInterval, //stores setInterval reference (used to check if hash changed)
213 		_frame, //iframe used for IE <= 7 
214 		_isLegacyIE = /msie (6|7)/.test(navigator.userAgent.toLowerCase()) && !+"\v1"; //feature detection based on Andrea Giammarchi's solution: http://webreflection.blogspot.com/2009/01/32-bytes-to-know-if-your-browser-is-ie.html

226 	/**
227 	 * Start listening/dispatching changes in the hash/history.
228 	 */
229 	Hasher.init = function(){
230 		var newHash = this.getHash();
231 		//TODO: use 'window.onhashchange' listener if browser supports it.
232 		if(_isLegacyIE){ //IE6 & IE7 [HACK]
233 			if(!_frame){
234 				_createFrame();
235 				_updateFrame();
236 			}
237 			_checkInterval = setInterval(_checkHistoryLegacyIE, 25);
238 		}else{ //regular browsers
239 			_checkInterval = setInterval(_checkHash, 25);
240 		}
241 		this.dispatchEvent(new HasherEvent(HasherEvent.INIT, _oldHash, newHash));
242 		_oldHash = newHash; //avoid dispatching CHANGE event just after INIT event (since it didn't changed).
243 	};
244 	
245 	/**
246 	 * Stop listening/dispatching changes in the hash/history.
247 	 */
248 	Hasher.stop = function(){
249 		clearInterval(_checkInterval);
250 		_checkInterval = null;
251 		this.dispatchEvent(new HasherEvent(HasherEvent.STOP, _oldHash, _oldHash)); //since it didn't changed oldHash and newHash should be the same.
252 	};

276 		separator = separator || '/';
277 		var hash = this.getHash(),
278 			regexp = new RegExp('^\\'+ separator +'|\\'+ separator +'$', 'g'); //match string starting and/or ending with separator
279 		hash = hash.replace(regexp, '');
280 		return hash.split(separator);
281 	};

364 		Hasher.dispatchEvent(new HasherEvent(HasherEvent.CHANGE, _oldHash, newHash));
365 		_oldHash = newHash;
366 	}

369 	 * Function that checks if hash has changed. [HACK]
370 	 * - used since most browsers don't dispatch the `onhashchange` event.

374 		var curHash = Hasher.getHash();
375 		if(curHash != _oldHash){

377 		}

381 	 * Check if browser history state has changed on IE <= 7. [HACK]
382 	 * - used since IE 6,7 doesn't generates new history record on hashchange.
383 	 * @private
384 	 */
385 	function _checkHistoryLegacyIE(){
386 		var windowHash = Hasher.getHash(),
387 			frameHash = _frame.contentWindow.frameHash;
388 		if(frameHash != windowHash && frameHash != _oldHash){ //detect changes made pressing browser history buttons. Workaround since history.back() and history.forward() doesn't update hash value on IE6/7 but updates content of the iframe.

390 			_dispatchChange(frameHash);
391 		}else if(windowHash != _oldHash){ //detect if hash changed (manually or using setHash)
392 			if(frameHash != windowHash){
393 				_updateFrame();
394 			}
395 			_dispatchChange(windowHash);
396 		}
397 	}

403 	function _createFrame(){
404 		_frame = document.createElement('iframe');

406 		_frame.style.display = 'none';
407 		document.body.appendChild(_frame);
408 	}

415 	function _updateFrame(){
416 		var frameDoc = _frame.contentWindow.document;
417 		frameDoc.open();
418 		frameDoc.write('<html><head><title>'+ Hasher.getTitle() +'</title><script type="text/javascript">var frameHash="'+ Hasher.getHash() +'";</script></head><body> </body></html>'); //stores current hash inside iframe.
419 		frameDoc.close();
420 	}

430 /**
431  * HasherEvent Object.
432  * <p>According to the HTML5 spec `hashchange` event should have `oldURL` and `newURL` properties, since the only portion of the URL that changes is the hash I decided to use `oldHash` and `newHash` instead. (http://www.whatwg.org/specs/web-apps/current-work/multipage/history.html#event-hashchange)</p>
433  * @param {String} eType	Hasher Event type.
434  * @param {String} oldHash	Previous Hash.
435  * @param {String} newHash	Current Hash.
436  * @constructor
437  */
438 var HasherEvent = function(eType, oldHash, newHash){
439 	/**
440 	 * Event Type
441 	 * @type String
442 	 */
443 	this.type = eType;
444 	/**
445 	 * Previous Hash value
446 	 * @type String
447 	 */
448 	this.oldHash = oldHash;
449 	/**
450 	 * Current Hash value
451 	 * @type String
452 	 */
453 	this.newHash = newHash;
454 };
455 
456 /**
457  * Returns string representation of the HasherEvent
458  * @return {String} A string representation of the object.
459  */
460 HasherEvent.prototype.toString = function(){
461 	return '[HasherEvent type="'+ this.type +'" oldHash="'+ this.oldHash +'" newHash="'+ this.newHash +'"]';
462 };
