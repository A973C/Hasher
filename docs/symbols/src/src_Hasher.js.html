 16 		_frame, //iframe used for IE <= 7 
 17 		_isLegacyIE = /msie (6|7)/.test(navigator.userAgent.toLowerCase()) && !+"\v1"; //feature detection based on Andrea Giammarchi's solution: http://webreflection.blogspot.com/2009/01/32-bytes-to-know-if-your-browser-is-ie.html

 22 	/**
 23 	 * Hasher
 24 	 * @class
 25 	 * @extends MM.EventDispatcher
 26 	 * @borrows MM.queryUtils.getQueryString as getQueryString
 27 	 * @borrows MM.queryUtils.getQueryObject as getQueryObject
 28 	 * @borrows MM.queryUtils.getParamValue as getParamValue
 29 	 * @borrows MM.queryUtils.hasParam as hasParam
 30 	 * @borrows MM.queryUtils.toQueryString as toQueryString
 31 	 */
 32 	this.Hasher = new MM.EventDispatcher();
 33 	
 34 	/**
 35 	 * Start listening/dispatching changes in the hash/history.
 36 	 */
 37 	Hasher.init = function(){
 38 		var newHash = Hasher.getHash();
 39 		//TODO: use 'window.onhashchange' listener if browser supports it.
 40 		if(_isLegacyIE){ //IE6 & IE7 [HACK]
 41 			if(!_frame){
 42 				_createFrame();
 43 				_updateFrame();
 44 			}
 45 			_checkInterval = setInterval(_checkHistoryLegacyIE, 50);
 46 		}else{ //regular browsers
 47 			_checkInterval = setInterval(_checkHash, 50);
 48 		}
 49 		this.dispatchEvent(new HasherEvent(HasherEvent.INIT, _oldHash, newHash));
 50 		_oldHash = newHash; //avoid dispatching CHANGE event just after INIT event (since it didn't changed).
 51 	};
 52 	
 53 	/**
 54 	 * Stop listening/dispatching changes in the hash/history.
 55 	 */
 56 	Hasher.stop = function(){
 57 		clearInterval(_checkInterval);
 58 		_checkInterval = null;
 59 		this.dispatchEvent(new HasherEvent(HasherEvent.STOP, _oldHash, _oldHash)); //since it didn't changed oldHash and newHash should be the same.
 60 	};

 62 	/**
 63 	 * Set Hash value.
 64 	 * @param {String} value	Hash value without '#'.
 65 	 */
 66 	Hasher.setHash = function(value){
 67 		location.hash = value;

 70 	/**
 71 	 * Return hash value as String.
 72 	 * @return {String}	Hash value without '#'.
 73 	 */
 74 	Hasher.getHash = function(){
 75 		return location.hash.substr(1);
 76 	};

 78 	/**
 79 	 * Return hash value as Array.
 80 	 * @param {String} [separator]	String used to divide hash (default = '/').

 82 	 */
 83 	Hasher.getHashAsArray = function(separator){
 84 		separator = separator || '/';
 85 		return Hasher.getHash().split(separator);
 86 	};

 89 	 * Set a new location or hash value without generating a history record for the current page. (user won't be able to return to current page)

 91 	 */

190 	/**
191 	 * Dispatch `HasherEvent.CHANGE` and stores hash value.
192 	 * @param {String} newHash	New Hash Value.
193 	 * @private
194 	 */
195 	function _dispatchChange(newHash){
196 		Hasher.dispatchEvent(new HasherEvent(HasherEvent.CHANGE, _oldHash, newHash));
197 		_oldHash = newHash;
198 	}
199 	
200 	/**
201 	 * Function that checks if hash has changed.
202 	 * - used since most browsers don't dispatch the `onhashchange` event.
203 	 * @private
204 	 */
205 	function _checkHash(){
206 		var curHash = Hasher.getHash();
207 		if(curHash != _oldHash){
208 			_dispatchChange(curHash);
209 		}
210 	}
211 	
212 	/**
213 	 * Check if browser history state has changed on IE <= 7. [HACK]
214 	 * - used since IE 6,7 doesn't generates new history record on hashchange.
215 	 * @private
216 	 */
217 	function _checkHistoryLegacyIE(){
218 		var windowHash = Hasher.getHash(),
219 			frameHash = _frame.contentWindow.frameHash;
220 		if(frameHash != windowHash && frameHash != _oldHash){ //detect changes made pressing browser history buttons. Workaround since history.back() and history.forward() doesn't update hash value on IE6/7 but updates content of the iframe.
221 			Hasher.setTitle(_frame.contentWindow.document.title);
222 			Hasher.setHash(frameHash);
223 			_dispatchChange(frameHash);
224 		}else if(windowHash != _oldHash){ //detect if hash changed (manually or using setHash)
225 			if(frameHash != windowHash){
226 				_updateFrame();
227 			}
228 			_dispatchChange(windowHash);
229 		}
230 	}
231 	
232 	/**
233 	 * Creates iframe used to record history state on IE <= 7. [HACK]
234 	 * @private
235 	 */
236 	function _createFrame(){
237 		_frame = document.createElement('iframe');
238 		_frame.src = 'javascript:false';
239 		_frame.style.display = 'none';
240 		document.body.appendChild(_frame);
241 	}
242 	
243 	/**
244 	 * Update iframe content, generating a history record and saving current hash/title on IE <= 7. [HACK]
245 	 * - based on Really Simple History, SWFAddress and YUI.history solutions.
246 	 * @private
247 	 */
248 	function _updateFrame(){
249 		var frameDoc = _frame.contentWindow.document;
250 		frameDoc.open();
251 		frameDoc.write('<html><head><title>'+ Hasher.getTitle() +'</title><script type="text/javascript">var frameHash="'+ Hasher.getHash() +'";</script></head><body> </body></html>'); //stores current title and current hash inside iframe.
252 		frameDoc.close();
253 	}
