 13 this.MM = this.MM || {};
 14 
 15 /**
 16  * Utilities for query string manipulation.
 17  * @namespace
 18  */
 19 MM.queryUtils = {
 20 	
 21 	/**
 22 	 * Gets full query as string with all special chars decoded.
 23 	 * @return {String}	Query string without starting '?'.
 24 	 */
 25 	getQueryString : function(){
 26 		return decodeURIComponent(location.search.substring(1));
 27 	},
 28 	
 29 	/**
 30 	 * Gets query as Object.
 31 	 * @return {Object}	Object with all the query "params => values" pairs.
 32 	 */
 33 	getQueryObject : function(){
 34 		var queryArr = this.getQueryString().split('&'), 
 35 			n = queryArr.length,
 36 			queryObj = {};
 37 		while (n--) {
 38 			queryArr[n] = queryArr[n].split('=');
 39 			queryObj[queryArr[n][0]] = queryArr[n][1];
 40 		}
 41 		return queryObj;
 42 	},
 43 	
 44 	/**
 45 	 * Get query parameter value.
 46 	 * @param {String} param	Parameter name.
 47 	 * @return {String}	Parameter value.
 48 	 */
 49 	getParamValue : function(param){
 50 		return this.getQueryObject()[param];
 51 	},
 52 	
 53 	/**
 54 	 * Checks if query contains parameter.
 55 	 * @param {String} param	Parameter name.
 56 	 * @return {Boolean} If parameter exist.
 57 	 */
 58 	hasParam : function(param){
 59 		return (this.getQueryString().indexOf(param+'=') >= 0);
 60 	},
 61 	
 62 	/**
 63 	 * Converts object into query string.
 64 	 * @param {Object} obj	Object with "params => values" pairs.
 65 	 * @return {String}	Formated query string starting with '?'.
 66 	 */
 67 	toQueryString : function(obj){
 68 		var query = [];
 69 		for(var param in obj){
 70 			query.push(param +'='+ obj[param]);
 71 		}
 72 		return (query.length)? '?'+ query.join('&') : '';
 73 	}
 74 	
 75 };
